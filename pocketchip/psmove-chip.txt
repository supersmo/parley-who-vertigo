How to build the kernel with hidraw support:

git clone --single-branch --branch debian/4.4.13-ntc-mlc --depth 1 https://github.com/NextThingCo/CHIP-linux.git
cd CHIP-linux

cp /boot/config-4.4.13-ntc-mlc .config
sed -i -e 's/# CONFIG_HIDRAW is not set/CONFIG_HIDRAW=y/' .config
echo "CONFIG_HID_LOGITECH_DJ=m" >>.config
make oldconfig

touch .scmversion  # http://stackoverflow.com/a/32699989/1047040
make LOCALVERSION="-ntc-mlc"

make modules_install # XXX OPTIONAL?!
cp arch/arm/boot/zImage /boot/
